name: Generate Apple Music Playlists (Manual)

on:
  workflow_dispatch:

permissions:
  contents: read
  actions: read

jobs:
  run:
    runs-on: ubuntu-latest

    steps:
    - uses: actions/checkout@v4

    - name: Set up Python
      uses: actions/setup-python@v5
      with:
        python-version: "3.11"

    - name: Install dependencies
      run: |
        python -m pip install --upgrade pip
        pip install -r requirements.txt

    - name: Run playlist generator
      env:
        SETLIST_API_KEY: ${{ secrets.SETLIST_API_KEY }}
        APPLE_DEVELOPER_TOKEN: ${{ secrets.APPLE_DEVELOPER_TOKEN }}
        APPLE_USER_TOKEN: ${{ secrets.APPLE_USER_TOKEN }}
        APPLE_STOREFRONT: ${{ secrets.APPLE_STOREFRONT }} # optional, default "us"
        USE_GOOGLE_SHEET: ${{ secrets.USE_GOOGLE_SHEET }}
        GOOGLE_SHEET_ID: ${{ secrets.GOOGLE_SHEET_ID }}
        GOOGLE_SERVICE_ACCOUNT_JSON: ${{ secrets.GOOGLE_SERVICE_ACCOUNT_JSON }}
      run: |
        # If you store events.csv in repo, ensure it's committed.
        python generate_playlists.py --events-file events.csv
